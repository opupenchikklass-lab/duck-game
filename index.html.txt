        body { background: #FFD700; display: flex; flex-direction: column; align-items: center; justify-content: center; height: 100vh; margin: 0; font-family: 'Arial', sans-serif; overflow: hidden; }
        #score { font-size: 50px; font-weight: bold; color: white; text-shadow: 2px 2px 5px rgba(0,0,0,0.3); margin-bottom: 20px; }

<body>
    <div id="score">0</div>
    <img id="duck" src="https://img.icons8.com/emoji/256/duck-emoji.png" alt="Duck">

    <script>
        const tg = window.Telegram.WebApp;
        tg.expand(); // Разворачиваем игру на весь экран

        let score = 0;
        const duck = document.getElementById('duck');
        const scoreDisplay = document.getElementById('score');

        duck.addEventListener('click', () => {
            score++;
            scoreDisplay.innerText = score;

            // Легкая вибрация телефона
            if (tg.HapticFeedback) {
                tg.HapticFeedback.impactOccurred('medium');
            }
        });
    </script>
